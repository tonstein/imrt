221 {
222 uint32_t n = _settings.numChannelsIn;
223 _in.resize({ n, nBufferFrames });
224
225 for (uint32_t frame = 0; frame < nBufferFrames; ++frame)
226 {
227 for (uint32_t channel = 0; channel < n; ++channel)
228 {
229 _in.getSample(channel, frame)
230 = ((float*)inputBuffer)[n * frame + channel];
231 }
232 }
233
234 uint32_t m = _settings.numChannelsOut;
235 _out.resize({ m, nBufferFrames });
236
237 int r = process(_in, _out, nBufferFrames);
238
239 for (uint32_t frame = 0; frame < nBufferFrames; ++frame)
240 {
241 for (uint32_t channel = 0; channel < m; ++channel)
242 {
243 ((float*)outputBuffer)[m * frame + channel]
244 = _out.getSample(channel, frame);
245 }
246 }
